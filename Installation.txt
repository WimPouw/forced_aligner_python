# second option Installing via pip for windows
https://montreal-forced-aligner.readthedocs.io/en/latest/installation.html

conda create -n aligner kaldi kalpy python=3.9

conda activate aligner

cd [scriptfolder]

git clone https://github.com/MontrealCorpusTools/Montreal-Forced-Aligner.git

cd Montreal-Forced-Aligner

pip install setuptools_scm

pip install -e .

python setup.py install

python setup.py develop

pip install pgvector==0.4.1

pip install textgrid==1.6.1

pip install pysoundfile==0.9.0

pip install webrtcvad==2.0.10

pip install soundfile==0.11.0

pip install ipykernel
